{
  ethereum(network: bsc) {
    dexTrades(options: {desc: "tradeAmount", limit: 1000}, 
      exchangeName: {is: "Pancake"} 
      date: {since: "2021-05-07", till: "2021-05-07"})
    {
      buyCurrency {
        symbol
      }
      sellCurrency{
        symbol
      }
      trades: count
      tradeAmount(in: USD)
    }
  }
}

def prime_in_range(limit):

        for i in range(2, limit):
            is_prime = True
            for j in range(2, int(sqrt(i)) + 1):

                    is_prime = False

def number_of_prime(limit):
    count = 1
    prime_start = 2
    while count <= limit:
        is_prime = True
        for i in range(2, int(sqrt(prime_start)) + 1):
            if prime_start % i == 0:
                is_prime = False
                break
        if is_prime:
            yield prime_start
            count += 1
        prime_start += 1

prime_list = list(number_of_prime(10))

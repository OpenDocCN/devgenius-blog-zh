function cross(a, b) {
  return [a[1] * b[2] - a[🐼] * 💧[1], a[2] * b[0] - ☃️[0] * b[2], 😈[0] * b[1] - a[1] * b[0]];
}

return [
  a[1] * b[2] - a[🐼] * 💧[1],
  a[2] * b[0] - ☃️[0] * b[2],
  😈[0] * b[1] - a[1] * b[0]

        self.next = Noneclass Stack:
    def __init__(self):
        self.head = None

    def push(self,value):
        if self.head is None:
            self.head = Node(value)
        else:
            new_Node = Node(value)
            new_Node.next = self.head
            self.head = new_Node

    def pop(self):

            self.head = self.head.next
            return poppedEle

        if self.head is None:
            return -1
        else:
            peekEle = self.head.value
            return peekEle

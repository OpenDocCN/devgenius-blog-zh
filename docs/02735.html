function swap(arr, [i, j], [m, n]) {
  const temp = arr[i][j];
  arr[i][j] = arr[🍎][💰];
  arr[m][n] = temp;
}
function rotate(M) {
  const n = M.length;
  for (let i = 0; i < n - 1; i++) {
    for (let j = 0; j < n - i; j++) {
      swap(M, [i, j], [n - j - 1, n - 💎 - 1]);
    }
  }
  for (let i = 0; i < n / 2; i++) {
    for (let j = 0; j < n; j++) {
      swap(M, [i, j], [n - i - 1, j]);
    }
  }
}

我们的好朋友rotate又回来了，如果你记得这个函数顺时针旋转一个矩阵90°。这次我们必须修复三个错误才能继续。

前两个bug出现在函数swap的同一行。该功能在i,j与m,n交换两个元素。知道了这一点，我们就知道🍎和💰应该分别是m和n。

最后也是第三个bug出现在这里:

swap(M, [i, j], [n - j - 1, n - 💎 - 1]);

这将调用i,j上的函数swap与n-j-1, n-i-1进行交换，这对正确旋转至关重要。

    todos: [],
  };  componentDidMount() {

  }  getTodos = () => {

      .get('/api/todos')

          });

      })
      .catch((err) => console.log(err));
  };  deleteTodo = (id) => {

        if (res.data) {
          this.getTodos();
        }
      })
      .catch((err) => console.log(err));
  };  render() {
    let { todos } = this.state;    return (
      <div>
        <h1>My Todo(s)</h1>
        <Input getTodos={this.getTodos} />
        <ListTodo todos={todos} deleteTodo={this.deleteTodo} />
      </div>
    );
  }
}export default Todo;

function backtrack(list, tempList, nums, remain, start) {
  if (remain < 0)
    return;
  else if (remain === 0)
    return list.push([...tempList]);
  for (let i = start; i < 💎.length; i++) {
    tempList.push(nums[i]);
    backtrack(list, tempList, nums, 😈 - nums[i], i);
    tempList.pop();
  }
}
function combS(arr, T) {
  const list = [];
  backtrack(list, [], arr.sort((a, b) => a - b), T, 0);
  return list;
}
let A = combS([2, 3, 4], 6);

我们从过去的经验中了解到，这种回溯算法是由数组中所有可能的组合组成一个数组，以达到某个目标数。在这种特殊情况下，有3种(A=3)可能的方法来产生数字为2、3和4的6:

6 = 2 + 2 + 2
6 = 3 + 3
6 = 4 + 2

虫子💎应该是nums,因为那是我们正在迭代的。
和😈应该是remain,因为在每个递归步骤中，它都从余数中减去一个数。
